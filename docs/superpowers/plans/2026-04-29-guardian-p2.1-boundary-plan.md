```python
"""Port policy: read from ~/.claude/contracts/port-policy.json."""
import json
import os

_CONTRACTS_DIR = os.path.join(os.path.expanduser("~"), ".claude", "contracts")
_PORT_POLICY_FILE = os.path.join(_CONTRACTS_DIR, "port-policy.json")


def _load_policy():
    with open(_PORT_POLICY_FILE, "r") as f:
        return json.load(f)


def _get_known_ports():
    policy = _load_policy()
    result = {}
    for port_str, info in policy.get("known_ports", {}).items():
        result[int(port_str)] = info
    return result


def _get_ghost_patterns():
    policy = _load_policy()
    return policy.get("ghost_patterns", [])


KNOWN_PORTS = _get_known_ports()
GHOST_PATTERNS = _get_ghost_patterns()
PROTECTED_PIDS = set()


def classify_port(port: int, bind_addr: str) -> dict:
    known = KNOWN_PORTS.get(port)
    is_public = bind_addr in ("0.0.0.0", "*", "::", "[::]")

    if known:
        if is_public and known["bind_policy"] == "localhost_only":
            return {"severity": "danger", "service": known["service"],
                    "reason": f"{known['service']} should bind localhost, got {bind_addr}"}
        if is_public and known["bind_policy"] == "allow_public":
            return {"severity": "info", "service": known["service"],
                    "reason": f"{known['service']} on {bind_addr} (ECS remote dev)"}
        return {"severity": "info", "service": known["service"], "reason": "normal"}

    if is_public:
        return {"severity": "warning", "service": "unknown",
                "reason": f"Unknown service on port {port} bound to {bind_addr}"}
    return {"severity": "info", "service": "unknown", "reason": f"Port {port} on {bind_addr}"}
```

```python
def collect_sessions(project_dir):
    registry_path = os.path.expanduser("~/.claude/hooks/state/session_registry.jsonl")
    if not os.path.exists(registry_path):
        return {"active_count": 0, "active_sessions": []}

    try:
        records = []
        with open(registry_path, "r", encoding="utf-8", errors="replace") as f:
            for line in f:
                line = line.strip()
                if not line:
                    continue
                try:
                    records.append(json.loads(line))
                except json.JSONDecodeError:
                    continue

        sessions = {}
        for r in records:
            sid = r.get("session_id")
            if not sid:
                continue
            if sid not in sessions:
                sessions[sid] = dict(r)
            else:
                sessions[sid].update(r)

        cutoff = datetime.now(TZ) - timedelta(minutes=5)
        active = []
        for s in sessions.values():
            if s.get("project_dir") != project_dir:
                continue
            if s.get("status") != "active":
                continue
            hb_str = s.get("heartbeat", "2000-01-01T00:00:00")
            try:
                hb = datetime.fromisoformat(hb_str)
                if hb.tzinfo is None:
                    hb = hb.replace(tzinfo=TZ)
                if hb >= cutoff:
                    active.append(s)
            except (ValueError, TypeError):
                continue

        return {
            "active_count": len(active),
            "active_sessions": [{"session_id": s.get("session_id", "?")[:8],
                                  "project": s.get("project_dir", "?"),
                                  "tool_calls": s.get("tool_calls", 0)}
                                 for s in active],
        }
    except Exception:
        return {"active_count": -1, "error": "session registry unavailable"}
```

```python
def collect_hook_events():
    log_path = os.path.expanduser("~/.claude/logs/hook-events.jsonl")
    if not os.path.exists(log_path):
        return {"recent_blocks": 0, "recent_warns": 0, "total_1h": 0}

    cutoff = (datetime.now(TZ) - timedelta(hours=1)).isoformat(timespec="seconds")
    blocks = warns = total = 0
    try:
        with open(log_path, "r") as f:
            for line in f:
                line = line.strip()
                if not line:
                    continue
                try:
                    e = json.loads(line)
                except json.JSONDecodeError:
                    continue
                ts = e.get("ts", "")
                if ts < cutoff:
                    continue
                ev = e.get("event", "")
                if ev == "allow":
                    continue
                total += 1
                if ev == "block":
                    blocks += 1
                elif ev == "warn":
                    warns += 1
    except OSError:
        pass

    return {"recent_blocks": blocks, "recent_warns": warns, "total_1h": total}
```

```python
def collect_unpushed():
    projects_dir = os.path.expanduser("~/projects")
    if not os.path.isdir(projects_dir):
        return []
    results = []
    for entry in sorted(os.listdir(projects_dir)):
        repo_path = os.path.join(projects_dir, entry)
        if not os.path.isdir(os.path.join(repo_path, ".git")):
            continue
        has_remote = run(["git", "-C", repo_path, "remote"])
        if not has_remote:
            continue
        count_str = run(["git", "-C", repo_path, "rev-list", "--count", "@{u}..HEAD"])
        if count_str and count_str.isdigit() and int(count_str) > 0:
            results.append({"repo": entry, "count": int(count_str)})
    return results


def collect_stale_docs(project_dir):
    import re as _re
    import time as _time
    plans_dir = os.path.join(project_dir, "docs", "plans")
    if not os.path.isdir(plans_dir):
        return []
    stale = []
    now = _time.time()
    for fn in os.listdir(plans_dir):
        if not fn.endswith("-design.md"):
            continue
        m = _re.match(r"(\d{4}-\d{2}-\d{2})", fn)
        if not m:
            continue
        try:
            with open(os.path.join(plans_dir, fn), "r", encoding="utf-8", errors="replace") as f:
                head = f.read(2000)
        except Exception:
            continue
        if "实现完成" in head or "pre-takeover" in head or "INVALID" in head:
            continue
        try:
            ts = _time.mktime(_time.strptime(m.group(1), "%Y-%m-%d"))
            days = int((now - ts) / 86400)
        except Exception:
            continue
        if days >= 14:
            short = fn.replace("-design.md", "").split("-", 3)[-1] if fn.count("-") >= 3 else fn
            stale.append({"file": fn, "short_name": short, "date": m.group(1), "days": days})
    return stale


def collect_build_freshness(project_dir):
    version_json = os.path.join(project_dir, "frontend", "dist", "version.json")
    if not os.path.exists(version_json):
        return {"exists": False}
    try:
        with open(version_json) as f:
            data = json.load(f)
        build_time = data.get("build_time", "")
        if not build_time:
            return {"exists": True, "age_hours": None}
        bt = datetime.fromisoformat(build_time.replace("Z", "+00:00"))
        age = (datetime.now(TZ) - bt).total_seconds() / 3600
        return {"exists": True, "build_time": build_time, "age_hours": round(age, 1)}
    except Exception:
        return {"exists": True, "age_hours": None}
```

```python
#!/usr/bin/env python3
"""Guardian P2.1 验证脚本。"""
import json
import os

STATE_DIR = os.path.expanduser("~/.claude/guardian")
CONTRACTS_DIR = os.path.expanduser("~/.claude/contracts")

def check(name, condition, detail=""):
    status = "PASS" if condition else "FAIL"
    print(f"  [{status}] {name}" + (f" — {detail}" if detail else ""))
    return condition

def main():
    ok = True
    print("=== Guardian P2.1 验证 ===\n")

    print("[契约层]")
    ok &= check("port-policy.json exists", os.path.exists(os.path.join(CONTRACTS_DIR, "port-policy.json")))
    ok &= check("hook-event.schema.json exists", os.path.exists(os.path.join(CONTRACTS_DIR, "hook-event.schema.json")))

    print("\n[解耦]")
    collector = os.path.join(STATE_DIR, "collector.py")
    with open(collector) as f:
        code = f.read()
    ok &= check("no import session_registry", "import session_registry" not in code)
    ok &= check("no sys.path hooks insert", 'sys.path' not in code or '.claude/hooks' not in code.replace("~", ""))

    print("\n[Snapshot]")
    with open(os.path.join(STATE_DIR, "latest.json")) as f:
        s = json.load(f)
    ok &= check("collector_version >= 0.2.0", s.get("collector_version", "") >= "0.2.0", s.get("collector_version"))
    ok &= check("has unpushed section", "unpushed" in s)
    ok &= check("has stale_docs section", "stale_docs" in s)
    ok &= check("has build_freshness section", "build_freshness" in s)
    ok &= check("hook_events has total_1h", "total_1h" in s.get("hook_events", {}))

    print("\n[health_context]")
    ctx_path = os.path.join(STATE_DIR, "health_context.json")
    ok &= check("health_context.json exists", os.path.exists(ctx_path))
    if os.path.exists(ctx_path):
        with open(ctx_path) as f:
            ctx = json.load(f)
        ok &= check("schema = guardian.health-context.v1", ctx.get("schema") == "guardian.health-context.v1")

    print("\n[port_policy]")
    pp = os.path.join(STATE_DIR, "port_policy.py")
    with open(pp) as f:
        pp_code = f.read()
    ok &= check("port_policy reads contracts JSON", "port-policy.json" in pp_code)
    ok &= check("no hardcoded KNOWN_PORTS dict literal", "9000:" not in pp_code)

    print("\n[Issue 类型]")
    codes = {i["issue_code"] for i in s.get("issues", [])}
    print(f"  活跃 issue 类型: {sorted(codes)}")

    print(f"\n{'=== ALL PASSED ===' if ok else '=== SOME FAILED ==='}")
    return 0 if ok else 1

if __name__ == "__main__":
    exit(main())
```
