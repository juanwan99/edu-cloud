# Knowledge ID Unification Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Eliminate the redundant KnowledgePoint UUID system; unify all knowledge references on ConceptGraphNode's semantic string IDs (`BIO_SR_CP_M1_CELL_THEORY`), extend the graph model to include 3-layer tree hierarchy (module→study_unit→concept), and enrich edge semantics from knowledge.db.

**Architecture:** ConceptGraphNode becomes the single knowledge backbone. sync_service projects modules (5) + study_units (99) + concepts (108) = 212 nodes with `contains` edges for tree hierarchy, plus the existing 335 concept-relation edges enriched with `evidence`/`pedagogical_use`. All consumer tables (QuestionKnowledgePoint, StudentKnowledgeMastery, StudentKnpMastery) switch FK from `knowledge_points.id` (UUID) to `concept_graph_nodes.id` (String). KnowledgePoint table and seed files are deleted.

**Tech Stack:** Python 3.11, SQLAlchemy 2.0 (async), Alembic, PostgreSQL, SQLite (knowledge.db read-only)

```python
def _sync_graph(session, data: dict) -> None:
    """Write full knowledge graph to PG (3-layer tree + concept relations)."""
    from edu_cloud.modules.knowledge_tree.models import (
        ConceptGraphNode, ConceptGraphEdge, ConceptBigConceptMap,
    )
    # Clear all projection tables
    session.execute(ConceptBigConceptMap.__table__.delete())
    session.execute(ConceptGraphEdge.__table__.delete())
    session.execute(ConceptGraphNode.__table__.delete())
    session.flush()

    now = datetime.utcnow()
    concepts = data["concepts"]
    study_units = data.get("study_units", [])
    edge_evidence = data.get("edge_evidence", {})

    # Collect all modules from concepts + study_units
    modules_seen = set()
    for c in concepts:
        if c.get("primary_module"):
            modules_seen.add(c["primary_module"])
    for su in study_units:
        if su.get("module"):
            modules_seen.add(su["module"])

    MODULE_NAMES = {
        "M1": "分子与细胞", "M2": "遗传与进化", "M3": "稳态与调节",
        "M4": "生态与环境", "M5": "生物技术与工程",
    }

    # 1. Module nodes
    for mod in sorted(modules_seen):
        session.add(ConceptGraphNode(
            id=f"mod:bio_sr:{mod}",
            name=MODULE_NAMES.get(mod, mod),
            knowledge_level="module",
            primary_module=mod,
            node_type="module",
            subject="biology",
            course_code="SW",
            display_order=int(mod[1:]) if mod[1:].isdigit() else 0,
            synced_at=now,
        ))

    # 2. Study unit nodes
    su_module_map = {}  # su_id → module
    su_concept_map = {}  # su_id → [concept_ids]
    for su in study_units:
        su_module_map[su["id"]] = su["module"]
        su_concept_map[su["id"]] = su["source_concept_ids"]
        m = su["module"]
        session.add(ConceptGraphNode(
            id=su["id"],
            name=su["name"],
            knowledge_level="study_unit",
            primary_module=m,
            description=su.get("description"),
            node_type="study_unit",
            subject="biology",
            course_code="SW",
            synced_at=now,
        ))

    # 3. Concept nodes (existing logic, enriched with subject/course_code)
    concept_ids = set()
    for c in concepts:
        concept_ids.add(c["id"])
        session.add(ConceptGraphNode(
            id=c["id"],
            name=c["name"],
            knowledge_level=c.get("knowledge_level", "L1"),
            primary_module=c.get("primary_module"),
            description=c.get("description"),
            node_type="concept",
            subject="biology",
            course_code="SW",
            difficulty=c.get("difficulty"),
            bloom_level=c.get("bloom_level"),
            aliases_json=c.get("aliases_json"),
            evidence_ids_json=c.get("evidence_ids_json"),
            review_status=c.get("review_status"),
            synced_at=now,
        ))

    # 4. BigConcept nodes + maps (existing logic, unchanged)
    for bc in data.get("big_concepts", []):
        session.add(ConceptGraphNode(
            id=bc["id"], name=bc["name"],
            knowledge_level="L1", primary_module=bc.get("primary_module"),
            description=bc.get("description"), node_type="big_concept",
            subject="biology", course_code="SW", synced_at=now,
        ))
    for m in data.get("bc_map", []):
        session.add(ConceptBigConceptMap(
            big_concept_id=m["big_concept_id"],
            concept_id=m["concept_id"],
        ))

    session.flush()

    # 5. Hierarchy edges (contains)
    # Module → Study Unit
    for su_id, mod in su_module_map.items():
        session.add(ConceptGraphEdge(
            source_id=f"mod:bio_sr:{mod}",
            target_id=su_id,
            relation_type="contains",
            strength=1.0,
            confidence=1.0,
        ))

    # Study Unit → Concept
    concepts_with_su = set()
    for su_id, cids in su_concept_map.items():
        for cid in cids:
            if cid in concept_ids:
                concepts_with_su.add(cid)
                session.add(ConceptGraphEdge(
                    source_id=su_id,
                    target_id=cid,
                    relation_type="contains",
                    strength=1.0,
                    confidence=1.0,
                ))

    # Orphan concepts → direct Module edge
    for cid in concept_ids - concepts_with_su:
        m_match = _MODULE_RE.search(cid)
        if m_match:
            mod = f"M{m_match.group(1)}"
            session.add(ConceptGraphEdge(
                source_id=f"mod:bio_sr:{mod}",
                target_id=cid,
                relation_type="contains",
                strength=1.0,
                confidence=1.0,
            ))

    # 6. Concept relation edges (enriched with evidence/pedagogical_use)
    for e in data["edges"]:
        if e["source_id"] in concept_ids and e["target_id"] in concept_ids:
            enrichment = edge_evidence.get(
                (e["source_id"], e["target_id"], e["relation_type"]), {}
            )
            session.add(ConceptGraphEdge(
                source_id=e["source_id"],
                target_id=e["target_id"],
                relation_type=e["relation_type"],
                strength=e.get("strength", 1.0),
                confidence=e.get("confidence", 1.0),
                review_status=e.get("review_status"),
                evidence=enrichment.get("evidence"),
                pedagogical_use=enrichment.get("pedagogical_use"),
            ))

    session.flush()
    logger.info(
        "synced knowledge graph: %d modules, %d study_units, %d concepts, edges=%d",
        len(modules_seen), len(study_units), len(concepts), session.query(ConceptGraphEdge).count() if hasattr(session, 'query') else '?',
    )
```

```python
Revision ID: {generated}

from alembic import op
import sqlalchemy as sa


def upgrade() -> None:
    # 1. question_knowledge_points: rename knowledge_point_id → concept_id
    #    Drop old FK, add new column, migrate data, drop old column
    op.add_column("question_knowledge_points",
                  sa.Column("concept_id", sa.String(64), nullable=True))

    # Migrate: UUID → concept_id via knowledge_points.code transformation
    # KnowledgePoint.code for concepts = "BIO_" + concept_id
    # So concept_id = substring(code, 5) — strip "BIO_" prefix
    op.execute("""
        UPDATE question_knowledge_points qkp
        SET concept_id = SUBSTRING(kp.code FROM 5)
        FROM knowledge_points kp
        WHERE kp.id = qkp.knowledge_point_id
          AND kp.level = 3
    """)
    # For level 2 (study units): code = "BIO_SU_M1_01" → su_id = "su:bio_sr:m1_01"
    # These are unlikely to exist in QKP but handle defensively
    op.execute("""
        UPDATE question_knowledge_points qkp
        SET concept_id = 'su:bio_sr:' || LOWER(REPLACE(SUBSTRING(kp.code FROM 5), '_', ':'))
        FROM knowledge_points kp
        WHERE kp.id = qkp.knowledge_point_id
          AND kp.level = 2
          AND qkp.concept_id IS NULL
    """)

    # Drop rows that couldn't be mapped (level=1 module nodes aren't real knowledge points)
    op.execute("DELETE FROM question_knowledge_points WHERE concept_id IS NULL")

    # Drop old FK and column
    op.drop_constraint("question_knowledge_points_knowledge_point_id_fkey",
                       "question_knowledge_points", type_="foreignkey")
```

```python
@tools.register(
    name="get_knowledge_tree",
    description=(
        "获取指定科目的知识点树。返回三层结构：模块→学习单元→概念。"
        "每个节点包含 id（语义化字符串如 BIO_SR_CP_M1_CELL_THEORY）、name、"
        "node_type（module/study_unit/concept）、primary_module、description。"
    ),

    domain="knowledge",
    parameters={
        "course_code": {"type": "string", "description": "科目代码: SW(生物)/SX(数学)", "default": "SW"},
        "module": {"type": "string", "description": "模块过滤: M1/M2/M3/M4/M5/all", "default": "all"},
    },
)
async def get_knowledge_tree(input: dict, ctx: ToolContext) -> ToolResult:
    from edu_cloud.modules.knowledge_tree.models import ConceptGraphNode, ConceptGraphEdge

    course_code = input.get("course_code", "SW")
    module_filter = input.get("module", "all")

    stmt = select(ConceptGraphNode).where(ConceptGraphNode.course_code == course_code)
    if module_filter != "all":
        stmt = stmt.where(ConceptGraphNode.primary_module == module_filter)

    result = await ctx.db.execute(stmt)
    nodes = result.scalars().all()

    # Build parent map from contains edges
    node_ids = [n.id for n in nodes]
    edges = await ctx.db.execute(
        select(ConceptGraphEdge.source_id, ConceptGraphEdge.target_id)
        .where(ConceptGraphEdge.relation_type == "contains",
               ConceptGraphEdge.target_id.in_(node_ids))
    )
    child_to_parent = {row[1]: row[0] for row in edges.all()}

    items = []
    for n in nodes:
        items.append({
            "id": n.id,
            "name": n.name,
            "node_type": n.node_type,
            "primary_module": n.primary_module,
            "description": n.description,
            "parent_id": child_to_parent.get(n.id),
        })

    return ToolResult(success=True, data={"knowledge_points": items, "total": len(items)})
```
